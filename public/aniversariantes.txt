    <script>
        let aniversariantesAtuais = [];
        let dadosEmpresa = {};
        let filtroAtual = 'hoje';

        async function carregarDadosUsuario() {
            const response = await fetch('/api/auth/me');
            const usuario = await response.json();
            document.getElementById('userName').textContent = usuario.nome || 'Usuário';
        }

        async function carregarEmpresa() {
            const response = await fetch('/api/empresa');
            dadosEmpresa = await response.json();
        }

        async function carregarStats() {
            const response = await fetch('/api/dashboard');
            const stats = await response.json();
            
            document.getElementById('totalHoje').textContent = stats.aniversariantes_hoje || 0;
            
            const responseMes = await fetch('/api/aniversariantes/mes/' + (new Date().getMonth() + 1));
            const aniversariantesMes = await responseMes.json();
            document.getElementById('totalMes').textContent = aniversariantesMes.length;
            
            const responseProximos = await fetch('/api/aniversariantes/proximos/30');
            const proximos = await responseProximos.json();
            document.getElementById('proximos').textContent = proximos.length;
        }

        async function filtrar(tipo) {
            filtroAtual = tipo;
            
            // Atualizar botões
            document.querySelectorAll('.filtro-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');

            let titulo = '';
            let dados = [];

            if (tipo === 'hoje') {
                const response = await fetch('/api/aniversariantes/hoje');
                dados = await response.json();
                titulo = '🎂 Aniversariantes de Hoje';
            } else if (tipo === 'mes') {
                const mes = new Date().getMonth() + 1;
                const response = await fetch(`/api/aniversariantes/mes/${mes}`);
                dados = await response.json();
                titulo = `📅 Aniversariantes de ${getNomeMes(mes)}`;
            } else if (tipo === 'proximos') {
                const response = await fetch('/api/aniversariantes/proximos/30');
                dados = await response.json();
                titulo = '⏰ Próximos 30 dias';
            } else {
                const response = await fetch('/api/clientes');
                dados = await response.json();
                titulo = '👥 Todos os Clientes';
            }

            aniversariantesAtuais = dados;
            document.getElementById('tituloLista').textContent = titulo;
            exibirLista(dados);
        }

        function getNomeMes(numero) {
            const meses = ['Janeiro', 'Fevereiro', 'Março', 'Abril', 'Maio', 'Junho', 
                          'Julho', 'Agosto', 'Setembro', 'Outubro', 'Novembro', 'Dezembro'];
            return meses[numero - 1];
        }

        function formatarData(dataISO) {
            if (!dataISO) return 'Não informada';
            const data = new Date(dataISO + 'T12:00:00');
            return data.toLocaleDateString('pt-BR');
        }

        function exibirLista(clientes) {
            const container = document.getElementById('containerLista');
            
            if (!clientes || clientes.length === 0) {
                container.innerHTML = `
                    <div class="sem-aniversarios">
                        <i class="fas fa-calendar-times"></i>
                        <h3>Nenhum aniversariante encontrado</h3>
                        <p>Não há clientes nesta lista.</p>
                    </div>
                `;
                return;
            }

            let html = '';
            clientes.forEach(cliente => {
                const dataNasc = cliente.data_nascimento ? formatarData(cliente.data_nascimento) : 'Não informada';
                const nomeEmpresa = dadosEmpresa.nome_empresa || 'nossa empresa';
                const mensagem = document.getElementById('mensagemPadrao').value
                    .replace('{empresa}', nomeEmpresa)
                    .replace('{cliente}', cliente.nome);

                html += `
                    <div class="aniversario-card">
                        <div class="nome">${cliente.nome}</div>
                        <div class="data">🎂 ${dataNasc}</div>
                        <div class="contato">
                            <i class="fas fa-phone"></i> ${cliente.telefone || 'Não informado'} <br>
                            <i class="fas fa-envelope"></i> ${cliente.email}
                        </div>
                        <a href="https://wa.me/${cliente.telefone?.replace(/\D/g, '') || ''}?text=${encodeURIComponent(mensagem)}" 
                           target="_blank" 
                           class="btn-parabens">
                            <i class="fab fa-whatsapp"></i> Enviar Parabéns
                        </a>
                    </div>
                `;
            });

            container.innerHTML = html;
        }

        function enviarParaTodos() {
            if (aniversariantesAtuais.length === 0) {
                alert('Nenhum aniversariante na lista atual');
                return;
            }

            const nomeEmpresa = dadosEmpresa.nome_empresa || 'nossa empresa';
            const mensagemBase = document.getElementById('mensagemPadrao').value;

            aniversariantesAtuais.forEach(cliente => {
                if (cliente.telefone) {
                    const mensagem = mensagemBase
                        .replace('{empresa}', nomeEmpresa)
                        .replace('{cliente}', cliente.nome);
                    
                    const numero = cliente.telefone.replace(/\D/g, '');
                    window.open(`https://wa.me/${numero}?text=${encodeURIComponent(mensagem)}`, '_blank');
                }
            });

            alert(`✅ Mensagens abertas para ${aniversariantesAtuais.length} clientes!`);
        }

        async function logout() {
            await fetch('/api/auth/logout', { method: 'POST' });
            window.location.href = '/login.html';
        }

        // Carregar dados iniciais
        carregarDadosUsuario();
        carregarEmpresa();
        carregarStats();
        filtrar('hoje');
    </script>
